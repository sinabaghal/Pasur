\newcommand{\drawsmallboxwitharrows}[6]{%
    % Assume we are inside a tikzpicture already
    \begin{scope}[shift={(#5)}]
      % Draw the box with fill color
      \node[draw, fill=#6, minimum width=.1cm, minimum height=.1cm] (box) at (#1,0) {#4};
      
      % Draw the arrows
      \foreach \col/\x in {#2} {
        \draw[\col, thick, midarrow] 
          (#1+\x,0.7+\x/2) 
          to[out=270,in=90] 
          (box.north);
      }
    \end{scope}
}
